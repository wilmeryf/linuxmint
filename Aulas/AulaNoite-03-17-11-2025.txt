#01) Configuração do Git Local
a) criar um novo repositório no Github
b) git clone https://github.com/seuusuario/seurepositório
c) git config --global user.name "Seu Nome"
d) git config --global user.email seuemail@seudominio.com

e) git config --global core.editor vim
f) git config --global merge.tool vimdiff
g) git config --global color.ui true

h) git config --list

i) git pull
   git add .
   git commit -m "Atualização"
   git push
